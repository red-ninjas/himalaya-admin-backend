import {
  Body,
  Controller,
  Delete,
  Get,
  Param,
  Patch,
  Post,
  Query,
  Version,
} from '@nestjs/common';
import { isArray } from 'lodash';
import { QueryService } from '../services/query.service';

import { ApiBearerAuth, ApiTags } from '@nestjs/swagger';
import {
  ComplexQueryRequest,
  PaginationMeta,
  QueryPaginationResource,
  SimpleQueryRequest,
} from '..';
import { AccessTokenRequired } from '../../auth/decorators/access-token-decorator';
import { EntityAccessorValidation } from '../decorators/entity-accessor.decorator';
import { applyFilter, applySort } from '../helpers';

@Controller({
  path: '_query/:entity',
  version: ['1'],
})
@EntityAccessorValidation()
@AccessTokenRequired()
@ApiTags('query')
@ApiBearerAuth()
export class QueryController {
  constructor(private readonly queryService: QueryService) {}

  @Post()
  @Version('1')
  create(@Param('entity') schema: string, @Body() createUserDto: any) {}

  @Post('list')
  @Version('1')
  async getRecords(
    @Param('entity') schema: string,
    @Body() options: ComplexQueryRequest,
  ): Promise<QueryPaginationResource<any>> {
    const fields = this.queryService.getMetaDataFields(schema);
    const repo = this.queryService.getRepository(schema);
    const queryBuilder = repo.createQueryBuilder();

    queryBuilder.skip(options.skip).take(options.take);

    const itemCount = await queryBuilder.getCount();
    const { entities } = await queryBuilder.getRawAndEntities();

    if (options.sort !== undefined && isArray(options.sort)) {
      for (let item of options.sort) {
        applySort(fields, item, queryBuilder);
      }
    }

    if (options.filter !== undefined && isArray(options.filter)) {
      for (let item of options.filter) {
        applyFilter(fields, item, queryBuilder);
      }
    }

    const pageMetaDto = new PaginationMeta({ itemCount, options });
    return new QueryPaginationResource(entities, pageMetaDto);
  }

  @Get('list')
  @Version('1')
  async getRecordsSimple(
    @Param('entity') schema: string,
    @Query() options: SimpleQueryRequest,
  ): Promise<QueryPaginationResource<any>> {
    const fields = this.queryService.getMetaDataFields(schema);
    const repo = this.queryService.getRepository(schema);
    const queryBuilder = repo.createQueryBuilder();

    queryBuilder.skip(options.skip).take(options.take);

    const itemCount = await queryBuilder.getCount();
    const { entities } = await queryBuilder.getRawAndEntities();

    const pageMetaDto = new PaginationMeta({ itemCount, options });
    return new QueryPaginationResource(entities, pageMetaDto);
  }

  createModel<T>(type: new () => T): T {
    return new type();
  }

  @Get()
  @Version('1')
  getSchema(@Param('entity') schema: string) {
    const avaiableFields = this.queryService.getMetaDataFields(schema);
    return avaiableFields;
  }

  @Get('id/:id')
  @Version('1')
  findOne(@Param('id') id: string, @Param('entity') schema: string) {
    return schema;
  }

  @Patch(':id')
  @Version('1')
  update(
    @Param('id') id: string,
    @Param('entity') schema: string,
    @Body() updateUserDto: any,
  ) {}

  @Delete(':id')
  @Version('1')
  remove(@Param('id') id: string, @Param('entity') schema: string) {}
}

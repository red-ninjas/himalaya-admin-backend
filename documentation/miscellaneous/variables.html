applyFilter
Default value : ( fields: ResourceDictonary<MetaDataResult>, field: FilterableRequest, queryBuilder: SelectQueryBuilder<any>, ) => { const foundField = fields[field.field]; if (foundField === undefined || foundField.isFilterable !== true) { throw new UnauthorizedException( field.field + ' is not accessable as filter field.', ); } else { const filter = {}; if (foundField.filterMatch == undefined) { foundField.filterMatch = FilterMatch.STARTS_OR_ENDS; } if (foundField.filterMatch == FilterMatch.STARTS_WITH) { filter[field.field] = ILike(`${field.term}%`); } else if (foundField.filterMatch == FilterMatch.EXACT) { filter[field.field] = `${field.term}`; } else if (foundField.filterMatch == FilterMatch.ENDS_WITH) { filter[field.field] = ILike(`%${field.term}`); } else if (foundField.filterMatch == FilterMatch.STARTS_OR_ENDS) { filter[field.field] = ILike(`%${field.term}%`); } queryBuilder.where(filter); } }
applySort
Default value : ( fields: ResourceDictonary<MetaDataResult>, field: SortableRequest, queryBuilder: SelectQueryBuilder<any>, ) => { const foundField = fields[field.field]; if (foundField === undefined || foundField.isSortable !== true) { throw new UnauthorizedException( field.field + ' is not accessable as order field.', ); } else { queryBuilder.orderBy(field.field, field.order); } }
